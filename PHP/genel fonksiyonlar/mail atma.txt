<?php
session_start();
use PHPMailer\PHPMailer\PHPMailer;
use PHPMailer\PHPMailer\SMTP;
use PHPMailer\PHPMailer\Exception;
 $_SESSION['mesaj']=$_POST['mesaj'];
 $_SESSION['konu']=$_POST['konu'];
// Load Composer's autoloader
require 'vendor/autoload.php';//composer phpmailer yklemesi
$mail = new PHPMailer(true);

try {
    //Server settings
    $mail->SMTPDebug =2;                      // Enable verbose debug output
    $mail->isSMTP();                                            // Send using SMTP
    $mail->Host       = 'smtp.gmail.com';                    // Set the SMTP server to send through
    $mail->SMTPAuth   = true;                                   // Enable SMTP authentication
    $mail->Username   = '';// SMTP username
    $mail->Password   = '';// SMTP password
    $mail->SMTPSecure = 'tls';         // Enable TLS encryption; `PHPMailer::ENCRYPTION_SMTPS` encouraged
    $mail->Port       = 587;                                    // TCP port to connect to, use 465 for `PHPMailer::ENCRYPTION_SMTPS` above

    //Recipients
    $mail->setFrom('mailadresi', 'Mailer');
    $mail->addAddress('kime atlacak', 'Joe User');     // Add a recipient

    // Content
    $mail->isHTML(true);                                  // Set email format to HTML
    $mail->Subject = $_SESSION['konu'];
    $mail->Body    = $_SESSION['user'].$_SESSION['mesaj'];


    $mail->send();
    echo 'Message has been sent';
    header("location:home.php")
} catch (Exception $e) {
    echo "Message could not be sent. Mailer Error: {$mail->ErrorInfo}";
}
?>
